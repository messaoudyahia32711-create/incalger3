    39→function AnimatedCounter({ target, duration = 2000 }: { target: number; duration?: number }) {
    40→  const [count, setCount] = useState(0);
    41→  const hasAnimated = useRef(false);
    42→  const startAnimation = useCallback(() => {
    43→    if (hasAnimated.current) return;
    44→    hasAnimated.current = true;
    45→    const start = Date.now();
    46→    const animate = () => {
    47→      const progress = Math.min((Date.now() - start) / duration, 1);
    48→      setCount(Math.floor((1 - Math.pow(1 - progress, 3)) * target));
    49→      if (progress < 1) requestAnimationFrame(animate);
    50→    };
    51→    requestAnimationFrame(animate);
    52→  }, [target, duration]);
    53→  useEffect(() => {
    54→    const timer = setTimeout(startAnimation, 800);
    55→    return () => clearTimeout(timer);
    56→  }, [startAnimation]);
    57→  return <span>{count.toLocaleString()}</span>;
    58→}

   716→function MessagingPanel({ locale, userId, userRole, otherPartyName }: { locale: Locale; userId: string; userRole: 'student' | 'admin'; otherPartyName: string }) {
   717→  const { messages, addMessage, markMessagesRead } = useAppStore();
   718→  const [newMsg, setNewMsg] = useState('');
   719→  const dir = localeDirection[locale];
   720→  const chatMessages = messages.filter((m: Message) => (m.from === userId || m.to === userId));
   721→  const otherRole = userRole === 'student' ? 'admin' : 'student';
   722→
   723→  useEffect(() => { markMessagesRead(userId, otherRole); }, [userId, otherRole, markMessagesRead]);
   724→
   725→  const handleSend = () => {
   726→    if (!newMsg.trim()) return;
   727→    addMessage({ id: `msg-${Date.now()}`, from: userId, fromName: userId === 'admin-001' ? 'المدير العام' : 'الطالب', fromRole: userRole, to: userRole === 'admin' ? 'student-001' : 'admin-001', text: newMsg, timestamp: new Date().toISOString(), read: false });
   728→    setNewMsg('');
   729→  };
   730→
   731→  return (
   732→    <div className="flex flex-col h-[500px]" style={{ direction: dir }}>
   733→      <div className="p-3 border-b border-slate-200 flex items-center gap-2"><div className="w-8 h-8 rounded-full bg-[#1B3A6B] flex items-center justify-center"><Shield className="w-4 h-4 text-white" /></div><span className="font-bold text-[#1B3A6B] text-sm">{otherPartyName}</span></div>
   734→      <div className="flex-1 overflow-y-auto p-4 space-y-3">
   735→        {chatMessages.length === 0 && <p className="text-center text-slate-400 text-sm py-8">{t(locale, 'common.noData')}</p>}
   736→        {chatMessages.map((msg: Message) => (
   737→          <div key={msg.id} className={`flex ${msg.from === userId ? 'justify-end' : 'justify-start'}`}>
   738→            <div className={`max-w-[80%] rounded-2xl px-4 py-2.5 text-sm ${msg.from === userId ? 'bg-[#1B3A6B] text-white rounded-br-sm' : 'bg-slate-100 text-slate-700 rounded-bl-sm'}`}>
   739→              <p>{msg.text}</p>
   740→              <p className={`text-[10px] mt-1 ${msg.from === userId ? 'text-white/50' : 'text-slate-400'}`}>{new Date(msg.timestamp).toLocaleTimeString(locale === 'ar' ? 'ar-DZ' : 'en', { hour: '2-digit', minute: '2-digit' })}</p>
   741→            </div>
   742→          </div>
   743→        ))}
   744→      </div>
   745→      <div className="p-3 border-t border-slate-200 flex gap-2">
   746→        <input type="text" value={newMsg} onChange={(e) => setNewMsg(e.target.value)} onKeyDown={(e) => e.key === 'Enter' && handleSend()} placeholder={t(locale, 'student.typeMessage')} className="flex-1 px-4 py-2.5 rounded-xl border-2 border-slate-200 focus:border-[#1B3A6B] outline-none text-sm" />
   747→        <button onClick={handleSend} className="px-4 py-2.5 rounded-xl bg-[#1B3A6B] text-white hover:bg-[#2952A3]"><Send className="w-4 h-4" /></button>
   748→      </div>
   749→    </div>
   750→  );
   751→}
