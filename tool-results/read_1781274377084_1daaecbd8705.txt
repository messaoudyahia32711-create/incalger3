    51→function AnimatedCounter({ target, duration = 2000 }: { target: number; duration?: number }) {
    52→  const [count, setCount] = useState(0);
    53→  const ref = useRef<HTMLSpanElement>(null);
    54→  const hasAnimated = useRef(false);
    55→
    56→  useEffect(() => {
    57→    if (hasAnimated.current) return;
    58→    const observer = new IntersectionObserver(
    59→      ([entry]) => {
    60→        if (entry.isIntersecting && !hasAnimated.current) {
    61→          hasAnimated.current = true;
    62→          const start = Date.now();
    63→          const animate = () => {
    64→            const elapsed = Date.now() - start;
    65→            const progress = Math.min(elapsed / duration, 1);
    66→            const eased = 1 - Math.pow(1 - progress, 3);
    67→            setCount(Math.floor(eased * target));
    68→            if (progress < 1) requestAnimationFrame(animate);
    69→          };
    70→          requestAnimationFrame(animate);
    71→        }
    72→      },
    73→      { threshold: 0.3 }
    74→    );
    75→    if (ref.current) observer.observe(ref.current);
    76→    return () => observer.disconnect();
    77→  }, [target, duration]);
    78→
    79→  return <span ref={ref}>{count.toLocaleString()}</span>;
    80→}
