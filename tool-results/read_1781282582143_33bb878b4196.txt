   776→function scoreCalc(base, netMonthly, breakEvenMonths, fixed, cashRows, yearNet){
   777→  let score = 0;
   778→  const revenue = Number(base?.rev || 0);
   779→  const variable = Number(base?.varCost || 0);
   780→  const marginRate = revenue > 0 ? (revenue - variable) / revenue : 0;
   781→  const minCash = Array.isArray(cashRows) && cashRows.length ? Math.min(...cashRows.map(r => Number(r.cumulative || 0))) : 0;
   782→
   783→  if(revenue > 0) score += 1.5;
   784→  if(netMonthly > 0) score += 2;
   785→  else if(netMonthly === 0 && revenue > 0) score += 0.5;
   786→
   787→  if(marginRate >= 0.5) score += 2;
   788→  else if(marginRate >= 0.3) score += 1.4;
   789→  else if(marginRate >= 0.15) score += 0.8;
   790→
   791→  if(breakEvenMonths > 0 && breakEvenMonths <= 6) score += 1.5;
   792→  else if(breakEvenMonths > 0 && breakEvenMonths <= 12) score += 1;
   793→  else if(breakEvenMonths > 0 && breakEvenMonths <= 18) score += 0.5;
   794→
   795→  if(revenue > 0){
   796→    const fixedRatio = fixed / revenue;
   797→    if(fixedRatio <= 0.30) score += 1;
   798→    else if(fixedRatio <= 0.50) score += 0.6;
   799→    else if(fixedRatio <= 0.80) score += 0.3;
   800→  }
   801→
   802→  if(minCash >= 0) score += 1;
   803→  else if(revenue > 0 && Math.abs(minCash) <= revenue * 2) score += 0.5;
   804→
   805→  if(yearNet > 0) score += 1;
   806→  else if(yearNet === 0 && revenue > 0) score += 0.3;
   807→
   808→  return Math.max(0, Math.min(10, score));
   809→}
